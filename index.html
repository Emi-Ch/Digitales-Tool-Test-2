    <style>

        body {

            display: flex;

            justify-content: center;

            align-items: center;

            height: 100vh;

            background-color: #f0f0f0;

            font-family: Arial, sans-serif;

            width: 100%;

<body>

    <h1>Digitales Tool im Chemieunterricht - Die Entstehung von temporären Dipol-Dipol-Wechselwirkungen</h1>

    <div id="sketch-container"></div>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/p5.js/1.4.0/p5.js"></script>

    <script>

        // Hier fügen Sie Ihren JavaScript-Code ein

        let t = 0;

        let animating = false;

        let secondMoleculeVisible = false;

        let firstMoleculePosition = 0;

        let displacementX = 0;

        let displacementY = 0;

        let displacementTargetX = 0;

        let displacementTargetY = 0;

        let autoMove = false;

        let moveInterval;

 

        function setup() {

            let canvas = createCanvas(windowWidth, windowHeight);

            canvas.parent('sketch-container');

            textAlign(CENTER, CENTER);

        }

 

        function draw() {

            background(255);

            drawFrame();

            drawSmallMolecule();

            drawStaticSmallElectronCloud();

            drawInstructionText();

 

            translate(width * 0.3, height / 2);

            drawMolecule();

            drawElectronCloud();

 

            if (displacementTargetX !== 0 || displacementTargetY !== 0) {

                displacementX += (displacementTargetX - displacementX) * 0.05;

                displacementY += (displacementTargetY - displacementY) * 0.05;

            }

 

            if (animating) {

                t += 0.16;

            }

 

            drawChargeCenters();

 

            if (secondMoleculeVisible) {

                translate(width * 0.4, 0);

                drawMolecule();

                drawElectronCloud();

                drawChargeCenters();

            }

        }

 

        function drawFrame() {

            stroke(0);

            strokeWeight(2);

            noFill();

            let frameWidth = width * 0.8;

            let frameHeight = height * 0.6;

            let xOffset = (width - frameWidth) / 2;

            let yOffset = (height - frameHeight) / 2;

            rect(xOffset, yOffset, frameWidth, frameHeight);

        }

 

        function drawInstructionText() {

            let instructionText = "Bewege die Wolke mit den Pfeiltasten";

            let secondInstructionText = "Klicke auf Molekül für Wechselwirkung";

            let thirdInstructionText = "Drücke 'A' für automatische Bewegung";

            let moleculeSize = 20;

            let instructionSize = moleculeSize / 2;

 

            fill(0);

            textSize(instructionSize);

            strokeWeight(0.5);

 

            text(instructionText, width * 0.2, height * 0.1);

            text(secondInstructionText, width * 0.2, height * 0.15);

            text(thirdInstructionText, width * 0.2, height * 0.2);

        }

 

        function drawMolecule(scaleFactor = 1) {

            push();

            scale(scaleFactor);

            stroke(0);

            strokeWeight(2);

            fill(0);

            textSize(20 * scaleFactor);

            text("C", -40, 0);

            text("C", 40, 0);

            textSize(16 * scaleFactor);

            text("H", -75, 0);

            text("H", -40, -35);

            text("H", -40, 35);

            text("H", 75, 0);

            text("H", 40, -35);

            text("H", 40, 35);

            strokeWeight(2 * scaleFactor);

            line(-30, 0, 30, 0);

            line(-60, 0, -48, 0);

            line(-40, -25, -40, -15);

            line(-40, 15, -40, 25);

            line(60, 0, 48, 0);

            line(40, -25, 40, -15);

            line(40, 15, 40, 25);

            pop();

        }

 

        function drawSmallMolecule() {

            push();

            translate(width / 2, height * 0.15);

            drawMolecule(0.5);

            pop();

        }

 

        function drawStaticSmallElectronCloud() {

            push();

            translate(width / 2, height * 0.15);

            scale(0.5);

            drawStaticElectronCloud();

            pop();

        }

 

        function drawStaticElectronCloud() {

            noFill();

            stroke(0);

            strokeWeight(1);

            beginShape();

            for (let angle = 0; angle < TWO_PI; angle += PI / 16) {

                let x = cos(angle) * 95;

                let y = sin(angle) * 65;

                vertex(x, y);

            }

            endShape(CLOSE);

        }

 

        function drawElectronCloud() {

            noFill();

            stroke(0);

            strokeWeight(2);

            beginShape();

            for (let angle = 0; angle < TWO_PI; angle += PI / 16) {

                let x = cos(angle) * 95 + displacementX;

                let y = sin(angle) * 65 + displacementY;

                vertex(x, y);

            }

            endShape(CLOSE);

        }

 

        function drawChargeCenters() {

            let offset = 10;

            let leftChargeX = -95 + displacementX + offset;

            let rightChargeX = 95 + displacementX - offset;

            let topChargeY = -60 + displacementY;

            let bottomChargeY = 60 + displacementY;

 

            if (displacementTargetX !== 0) {

                displacementY = 0;

            }

 

            if (displacementY > 0) {

                fill(0, 0, 255);

                noStroke();

                textSize(16);

                text("δ-", 0, bottomChargeY);

                fill(255, 0, 0);

                text("δ+", 0, topChargeY);

            } else if (displacementY < 0) {

                fill(0, 0, 255);

                noStroke();

                textSize(16);

                text("δ-", 0, topChargeY);

                fill(255, 0, 0);

                text("δ+", 0, bottomChargeY);

            } else {

                let movingRight = displacementX > 0;

                if (movingRight) {

                    fill(0, 0, 255);

                    noStroke();

                    textSize(16);

                    text("δ-", rightChargeX, 0);

                    fill(255, 0, 0);

                    text("δ+", leftChargeX, 0);

                } else {

                    fill(0, 0, 255);

                    noStroke();

                    textSize(16);

                    text("δ-", leftChargeX, 0);

                    fill(255, 0, 0);

                    text("δ+", rightChargeX, 0);

                }

            }

        }

 

        function keyPressed() {

            if (keyCode === LEFT_ARROW) {

                displacementTargetX = -20;

            } else if (keyCode === RIGHT_ARROW) {

                displacementTargetX = 20;

            } else if (keyCode === UP_ARROW) {

                displacementTargetY = -20;

            } else if (keyCode === DOWN_ARROW) {

                displacementTargetY = 20;

            }

            if (key === ' ' && !secondMoleculeVisible) {

                animating = !animating;

            }

            if (key === 'A' || key === 'a') {

                autoMove = !autoMove;

                if (autoMove) {

                    moveInterval = setInterval(randomMove, 333);

                } else {

                    clearInterval(moveInterval);

                    displacementTargetX = 0;

                    displacementTargetY = 0;

                }

            }

        }

 

        function keyReleased() {

            if (!autoMove) {

                displacementTargetX = 0;

                displacementTargetY = 0;

            }

        }

 

        function mousePressed() {

            let d = dist(mouseX, mouseY, width / 2, height * 0.15);

            if (d < 50) {

                animating = false;

                secondMoleculeVisible = true;

            }

        }

 

        function randomMove() {

            let moves = [

                { x: -20, y: 0 },

                { x: 20, y: 0 },

                { x: 0, y: -20 },

                { x: 0, y: 20 }

            ];

            let move = random(moves);

            displacementTargetX = move.x;

            displacementTargetY = move.y;

        }
